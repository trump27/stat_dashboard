指数対策

scientific_notation <- function(x) {
     x <- format(x, scientific = TRUE)
     x <- gsub("^(.*)e", "'\\1'e", x)
     x <- gsub("e", "%*%10^", x)
     x <- gsub('\\+', '', x)
     parse(text = x)
}

描画

d <- function(csv, title, yscale=FALSE) {
  df <- read.csv(csv)
  p <- ggplot(df, aes(x=as.Date(date), y=value)) +
    geom_line() +
    geom_smooth(method=loess) +
    labs(
      title = title,
      x = "Date",
      y = "Amount"
  )
  if (yscale) {
    p <- p + scale_y_continuous(label=scientific_notation)
  }
  print(p)
}
